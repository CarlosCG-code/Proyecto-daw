import java.util.*;

public class Ejelsisio8tema2 {
    public static void main(String[] args) {
        Random random = new Random();
        Scanner scanner = new Scanner(System.in);
        int numero1 = random.nextInt(99) + 1;
        int numero2 = random.nextInt(99) + 1;
        System.out.println("Suma los siguientes números:");
        System.out.println(numero1 + " + " + numero2 + " = ?");
        System.out.print("Tu respuesta: ");
        int respuestaUsuario = scanner.nextInt();
        int sumaCorrecta = numero1 + numero2;
        if (respuestaUsuario == sumaCorrecta) {
            System.out.println("La suma es " + sumaCorrecta);
        } else {
            System.out.println("La respuesta correcta era " + sumaCorrecta);
        }
    }
}
