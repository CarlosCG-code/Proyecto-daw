package com.mycompany.proyecto;
import java.util.Scanner;
/**
 * Math.sqrt()
 * @author usuario
 */
public class Ejelsisio2tema4 {
    static void calcularDistancia(double x1, double x2, double y1, double y2){
        double distancia;
        distancia = Math.sqrt((Math.pow(x1, 2) - Math.pow(x2, 2)) + (Math.pow(y1, 2) - Math.pow(y2, 2)));
        System.out.println("La distancia es: " + distancia);
    }

    public static void main(String[] args){
    Scanner sc = new Scanner(System.in);
    double x1, x2, y1, y2;
    System.out.print("Dame tu primera variable(x1): ");
    x1 = sc.nextDouble();
    System.out.print("Dame tu primera variable(x2): ");
    x2 = sc.nextDouble();
    System.out.print("Dame tu primera variable(y1): ");
    y1 = sc.nextDouble();
    System.out.print("Dame tu primera variable(y2): ");
    y2 = sc.nextDouble();
    calcularDistancia(x1, x2, y1, y2);
    }
}
